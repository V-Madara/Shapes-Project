Custom Shapes Example

import turtle
s = turtle.Screen()
s.addshape("rocket", (
    (-5, -10), (0, 20), (5, -10), 
    (0, -10), (0, -20)
))
t = turtle.Turtle()
t.shape("rocket")
t.shapesize(2)
t.forward(100)
